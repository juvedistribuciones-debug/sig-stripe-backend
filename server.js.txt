import express from "express";
import Stripe from "stripe";
import cors from "cors";

const app = express();
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY); // usa la variable de entorno
app.use(cors());
app.use(express.json());

// Crear sesiÃ³n de Stripe Checkout
app.post("/create-checkout-session", async (req, res) => {
  try {
    const { selectedDate, selectedHours } = req.body;

    if (!selectedDate || !selectedHours || selectedHours.length === 0) {
      return res.status(400).json({ error: "Datos incompletos" });
    }

    const totalHours = selectedHours.length;

    const session = await stripe.checkout.sessions.create({
      payment_method_types: ["card"],
      line_items: [
        {
          price_data: {
            currency: "usd",
            product_data: {
              name: `Reserva SIG Sportsplex - ${selectedDate}`,
              description: `Horas reservadas: ${selectedHours.join(", ")}`,
            },
            unit_amount: 160 * 100, // $160 por hora
          },
          quantity: totalHours,
        },
      ],
      mode: "payment",
      success_url: "https://tusitio.wixsite.com/success",  // cambia por tu URL
      cancel_url: "https://tusitio.wixsite.com/cancel",     // cambia por tu URL
    });

    res.json({ url: session.url });
  } catch (error) {
    console.error("Error creando sesiÃ³n:", error);
    res.status(500).json({ error: "Error creando sesiÃ³n de pago" });
  }
});

app.get("/", (req, res) => {
  res.send("Servidor Stripe funcionando ðŸš€");
});

app.listen(4242, () => console.log("âœ… Servidor corriendo en puerto 4242"));
